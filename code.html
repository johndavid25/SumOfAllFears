    function FindSum(numbers, answer) {
      let checkValue = 0;
      let found = false;
      for (let i = 0; i <= numbers.length - 1; i++) {
          checkValue = answer - numbers[i];
          if (numbers.includes(checkValue, i + 1)) {
              found = true;
              break;
          }
      }
      return found;
  }
